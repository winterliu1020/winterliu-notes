基础操作 

{
    // "query": {
    //     "match": {
    //         "category": "米"
    //     }
    // }
    "query": {
        "match_all": {

        },
        "match_phrase": { // 完全匹配
            "category": "小华" // 无结果
        }
    },
    "highlight": { // 对查询结果的category这个字段高亮显示
        "fields": {
            "category": {}
        }
    },

    // aggs是单独出现一个统计数据在查询结果中，其它的原始数据还会显示出来，加上size=0原始数据就不显示
    "aggs": { // 聚合操作，其实就是统计结果
        "price_group": { // 分组名称，随意起的，按照价格分组
            "terms": { // 分组
                "fields": "price" // 分组字段
            }
        },
        "price_avg": { // 价格平均值
            "avg": { // 平均值
                "fields": "price"
            }
        }
    }, // 还会出现原始数据

    // 当前页;分页查询 from是：（页码数-1）*每一页个数
    "from": 0,
    "size": 3,
    "_source": ["title"], // 对于想要的数据指定查询
    "sort": {
        "price": {
            "order": "desc"
        }
    }
}

{
    "query": {
        "bool": { // 表示条件
            // "must": [ 
            //     // 表示多个条件必须同时成立
            //     {
            //         "match": {
            //             "category": "小米"
            //         }
            //     },
            //     {
            //         "match": {
            //             "price": "389"
            //         }
            //     }
            // ]


            "should": [ 
                // 表示多个条件or
                {
                    "match": {
                        "category": "小米"
                    }
                },
                {
                    "match": {
                        "category": "华为"
                    }
                }
            ],
            "filter": { // 过滤
                "range": { // 范围查询
                    "price": {
                        "gt": 400
                    }
                }
            }
        }
    }
}

            "type": "keyword", // 不能被分词

        },
